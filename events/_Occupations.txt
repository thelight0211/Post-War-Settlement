	immediate = {log = "[GetDateText]: [Root.GetName]: event sov.partition.1"}
	title = ""
	desc = sov.partition.1.d
	
	hidden = yes
	
	trigger = {
		SOV = { exists = no }
		NOT = { has_country_flag = conquest_SOV }
		NOT = { has_country_flag = nocontrol_SOV }
	}
	
	mean_time_to_happen = { days = 1 }
	fire_only_once = no
	immediate = {
		set_country_flag = { flag = sov_state_count value = 0 }
		every_owned_state = {
			limit = {
				is_core_of = SOV
				is_on_continent = europe
				NOT = { is_core_of = ROOT }
            }
			ROOT = { 
				modify_country_flag = { flag = sov_state_count value = 1 } 
				set_country_flag = conquest_SOV
			}
        }
		IF = {
			limit = {
				NOT = { has_country_flag = conquest_SOV }
			}
			set_country_flag = nocontrol_SOV
		}
	}
	
	option = {
		name = "Trigger has fired"
	}
}
